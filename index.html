<!-- ✅ Script ปุ่ม Play AR พร้อม catch error -->

<script>

   function triggerAR(id) {

     const viewer = document.getElementById(id);

     if (viewer && viewer.activateAR) {

       viewer.activateAR().catch((error) => {

         alert("AR ไม่พร้อมบนอุปกรณ์นี้ หรือถูกบล็อกโดย browser\n(แนะนำให้กดปุ่ม AR บนมุมขวาของโมเดลแทน)");

         console.warn("activateAR failed:", error);

       });

     }

   }

</script>
